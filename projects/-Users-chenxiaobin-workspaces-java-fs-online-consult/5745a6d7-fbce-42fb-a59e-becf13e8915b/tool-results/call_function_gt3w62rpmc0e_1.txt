   516→  static prependQueryParameter(url, newParam) {
   517→
   518→    if(url && url.includes("proj/page/marketing-page")) {
   519→      return url
   520→    }
   521→
   522→    const [baseUrl, queryString] = url.split("?");
   523→
   524→    const urlParams = queryString.split("&").reduce((result, str) => {
   525→      const [key, value] = str.split("=");
   526→      result[key] = value;
   527→      return result;
   528→    }, {});
   529→
   530→    const resourceUrl = decodeURIComponent(urlParams.resourceUrl);
   531→    // 分割基础URL和查询字符串
   532→    const [baseUrl1, queryString1] = resourceUrl.split("?");
   533→    // 若没有查询字符串，直接添加
   534→
   535→    const targetWithEqual = `apiName=`;
   536→    const paramToAdd = `&${newParam}`;
   537→    const regex = new RegExp(`(${targetWithEqual}[^&]*)`, "i");
   538→
   539→    const decodeResourceUrl = queryString1.includes(targetWithEqual)
   540→      ? `${baseUrl1}?${queryString1.replace(regex, `$1${paramToAdd}`)}`
   541→      : `${baseUrl1}?${queryString1}`;
   542→
   543→    const encodedNewResourceUrl = encodeURIComponent(decodeResourceUrl);
   544→
   545→    // 构建正则表达式匹配原resourceUrl参数
   546→    const resourceUrlPattern = /resourceUrl=[^&]*/;
   547→    const newResourceUrlParam = `resourceUrl=${encodedNewResourceUrl}`;
   548→
   549→    // 替换参数
   550→    if (resourceUrlPattern.test(url)) {
   551→      return url.replace(resourceUrlPattern, newResourceUrlParam);
   552→    } else {
   553→      // 若不存在resourceUrl参数，则追加到URL末尾
   554→      const separator = url.includes("?") ? "&" : "?";
   555→      return `${url}${separator}resourceUrl=${encodedNewResourceUrl}`;
   556→    }
   557→  }
